给一个满足下述两条属性的 m x n 整数矩阵：  
每行中的整数从左到右按非严格递增顺序排列。  
每行的第一个整数大于前一行的最后一个整数。  

给一个整数 target ，如果 target 在矩阵中，返回 true ；否则，返回 false   

示例：  
    ↓  
1   3   5   7  
10  11  16  20  
23  30  34  60  

输入：matrix = [[1,3,5,7],[10,11,16,20],[23,30,34,60]],   target = 3  
输出：true  

***矩阵展开***

我的解法一：最后一个<= target  
先把每行最大值存入数组perLineMax中，接下来找出最后一个 <= target的元素，它的+1就是target可能在的行。考虑到target大于所有和小于所有的特殊情况，要在开头处理。找到在哪行后就普通查找

错因：没考虑到边界情况，只有一行，只有一行一个，只有两个且独自一行，行末刚好是目标值

```cpp
class Solution
{
public:
    bool searchMatrix(vector<vector<int>> &matrix, int target)
    {
        int m = matrix.size(); // m行
        int n = matrix[0].size(); // n列

        // 因为high停在最后一个 <= target 的位置，
        // 因此target大于所有，high = m-1；当target小于所有，high = -1，
        // target小于最小值
        if (target < matrix[0][0]) return 0;
        // target大于最大值
        if (target > matrix[m-1][n-1]) return 0;

        vector<int> perLineMax; // 每行最后一列--最大值
        for (int i = 0; i < m; i++)
        {
            perLineMax.push_back(matrix[i][n - 1]);
        }

        // 找target可能在哪行
        // 找最后一个 <= target 的元素
        int line;
        int low = 0;
        int high = m - 1;
        int mid;
        while (low <= high)
        {
            mid = low + (high - low) / 2;
            if (perLineMax[mid] < target)
            {
                low = mid + 1;
            }
            else if (perLineMax[mid] > target)
            {
                high = mid - 1;
            }
            else 
            {
                return 1;
            }
        }
        // high停在最后一个 <= target 的位置，
        // 我们要找的line是最后一个 <= target 的后一个，需要 +1
        line = high + 1;
        
        int l = 0;
        int r = n - 1;
        while (l <= r)
        {
            mid = l + (r - l) / 2;
            if (matrix[line][mid] == target)
            {
                return 1;
            }
            else if (matrix[line][mid] < target)
            {
                l = mid + 1;
            }
            else
            {
                r = mid - 1;
            }
        }
        return 0;
    }
};
```

官方思路：
直接从矩阵两角开始收缩，利用m或n的性质结合mid确定行和列

```cpp
class Solution
{
public:
    bool searchMatrix(vector<vector<int>> &matrix, int target)
    {
        int m = matrix.size(), n = matrix[0].size();
        int left = 0, right = m * n - 1; // 闭区间 [left, right]

        while (left <= right)
        {
            int mid = left + (right - left) / 2;
            int x = matrix[mid / n][mid % n];
            // 每行长度为n，mid / n 就是所在行
            // 每从 0 开始凑成一次 n-1 就算一行，所以mid % n 把完整的n个给去掉求出列
            // 同理可以用 m 表示

            if (x == target)
            {
                return true;
            }
            else if (x < target)
            {
                left = mid + 1; // 收缩左边界
            }
            else
            {
                right = mid - 1; // 收缩右边界
            }
        }
        return false;
    }
};
```
